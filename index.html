""" Bhati March 2026 — Railway Server Runs fetch every 60 seconds + serves data.json via HTTP """ import threading import time import json import os from http.server import HTTPServer, BaseHTTPRequestHandler from datetime import datetime, timezone DATA_FILE = "data.json" FETCH_INTERVAL = 60 class Handler(BaseHTTPRequestHandler): def do_OPTIONS(self): # Browser preflight request — must respond with CORS headers self.send_response(200) self._cors() self.end_headers() def do_GET(self): path = self.path.split("?")[0] # strip query params like ?t=123 if path in ("/", "/data.json"): self._serve_json() elif path == "/health": self._send(200, "OK", "text/plain") else: self._send(404, "Not found", "text/plain") def _cors(self): self.send_header("Access-Control-Allow-Origin", "*") self.send_header("Access-Control-Allow-Methods", "GET, OPTIONS") self.send_header("Access-Control-Allow-Headers", "Content-Type") def _serve_json(self): if os.path.exists(DATA_FILE): with open(DATA_FILE, "r") as f: content = f.read() self.send_response(200) self.send_header("Content-Type", "application/json") self.send_header("Cache-Control", "no-cache, no-store") self._cors() self.end_headers() self.wfile.write(content.encode()) else: self._send(503, '{"error":"Data not ready yet, please wait 60s..."}', "application/json") def _send(self, code, body, ctype): self.send_response(code) self.send_header("Content-Type", ctype) self._cors() self.end_headers() self.wfile.write(body.encode()) def log_message(self, format, *args): pass # suppress noisy request logs def fetch_loop(): print(f"[{datetime.now().strftime('%H:%M:%S')}] Fetch loop started — every {FETCH_INTERVAL}s") # Small delay on first run so server is up before fetch starts time.sleep(3) while True: try: print(f"[{datetime.now().strftime('%H:%M:%S')}] Running fetch...") import fetch_data import importlib importlib.reload(fetch_data) print(f"[{datetime.now().strftime('%H:%M:%S')}] Fetch complete.") except Exception as e: print(f"[{datetime.now().strftime('%H:%M:%S')}] Fetch error: {e}") time.sleep(FETCH_INTERVAL) if __name__ == "__main__": PORT = int(os.environ.get("PORT", 8080)) fetch_thread = threading.Thread(target=fetch_loop, daemon=True) fetch_thread.start() print(f"Server running on port {PORT}") server = HTTPServer(("0.0.0.0", PORT), Handler) server.serve_forever()
