""" Bhati March 2026 — Fetch & Analyse Optimized for LOW Firestore reads """ import firebase_admin from firebase_admin import credentials, firestore import pandas as pd from datetime import datetime, timezone, timedelta import json, os, math, tempfile # ══════════════════════════════════════════════════════════════════════════════ # CONFIG # ══════════════════════════════════════════════════════════════════════════════ SCAN_COLLECTION = "Bhati-March-2026" USER_COLLECTION = "User" LAST_FETCH_FILE = "last_fetch.json" USER_CACHE_FILE = "user_cache.json" SEWADAR_EXCEL = os.environ.get("SEWADAR_EXCEL_PATH", "Sewadar Details.xlsx") OUTPUT_JSON = "data.json" TIMESTAMP_FIELD = "timestamp" USER_CACHE_HOURS = 168 # 7 DAYS - users don't change often MAX_DAYS_KEEP = 7 # Only keep last 7 days of scans in memory IST = timedelta(hours=5, minutes=30) # ══════════════════════════════════════════════════════════════════════════════ # FIREBASE INIT # ══════════════════════════════════════════════════════════════════════════════ if not firebase_admin._apps: secret = os.environ.get("FIREBASE_SERVICE_ACCOUNT") if secret: sa = json.loads(secret) with tempfile.NamedTemporaryFile(mode="w", suffix=".json", delete=False) as f: json.dump(sa, f) tmp_path = f.name cred = credentials.Certificate(tmp_path) else: cred = credentials.Certificate( r"C:\Users\verma\Desktop\Dashboards\USER - Gameplay\service_account.json" ) firebase_admin.initialize_app(cred) db = firestore.client() # ══════════════════════════════════════════════════════════════════════════════ # SEWADAR EXCEL LOOKUP (only loads once, cached in memory) # ══════════════════════════════════════════════════════════════════════════════ badge_lookup = {} if os.path.exists(SEWADAR_EXCEL): df_sew = pd.read_excel(SEWADAR_EXCEL) df_sew["Badge Number"] = df_sew["Badge Number"].astype(str).str.strip() for _, row in df_sew.iterrows(): badge_lookup[row["Badge Number"]] = { "name": str(row.get("Name of Sewadar", "")).strip(), "gender": str(row.get("Gender", "")).strip(), "satsang_point": str(row.get("Satsang Point", "")).strip(), "centre": str(row.get("Centre", "")).strip(), "department": str(row.get("Deployed Department", "")).strip(), } print(f" Badge lookup: {len(badge_lookup)} sewadars from Excel") def lookup_badge(badge_no): return badge_lookup.get(str(badge_no).strip(), { "name":"Unknown","gender":"","satsang_point":"","centre":"","department":"" }) # ══════════════════════════════════════════════════════════════════════════════ # USER CACHE - ONLY FETCH ONCE PER WEEK (huge read savings!) # ══════════════════════════════════════════════════════════════════════════════ def load_user_cache(): if os.path.exists(USER_CACHE_FILE): with open(USER_CACHE_FILE) as f: cache = json.load(f) cached_at = datetime.fromisoformat(cache.get("cached_at","2000-01-01")) if cached_at.tzinfo is None: cached_at = cached_at.replace(tzinfo=timezone.utc) age_h = (datetime.now(timezone.utc) - cached_at).total_seconds() / 3600 if age_h < USER_CACHE_HOURS: print(f" Using cached user data ({age_h:.1f}h old)") return cache.get("users", {}) return None def fetch_and_cache_users(): print(" REFRESHING user cache from Firestore (only happens once/week!)...") user_map = {} for doc in db.collection(USER_COLLECTION).stream(): d = doc.to_dict() user_map[doc.id] = { "displayName": d.get("displayName",""), "badgeNumber": d.get("badgeNumber",""), } with open(USER_CACHE_FILE, "w") as f: json.dump({"cached_at": datetime.now(timezone.utc).isoformat(), "users": user_map}, f) print(f" Cached {len(user_map)} users.") return user_map # Only fetches from Firestore if cache is >7 days old user_cache = load_user_cache() or fetch_and_cache_users() userid_to_badge = {uid: info.get("badgeNumber","") for uid, info in user_cache.items()} # ══════════════════════════════════════════════════════════════════════════════ # TIMESTAMP HELPERS # ══════════════════════════════════════════════════════════════════════════════ def load_last_fetch(): if os.path.exists(LAST_FETCH_FILE): with open(LAST_FETCH_FILE) as f: dt = datetime.fromisoformat(json.load(f)["last_fetch"]) return dt.replace(tzinfo=timezone.utc) if dt.tzinfo is None else dt return None def save_last_fetch(dt): with open(LAST_FETCH_FILE,"w") as f: json.dump({"last_fetch": dt.isoformat()}, f, indent=2) def to_ist_naive(ts): if hasattr(ts, "tzinfo") and ts.tzinfo is not None: from datetime import timezone as tz ist_tz = tz(IST) return ts.astimezone(ist_tz).replace(tzinfo=None) return ts # ══════════════════════════════════════════════════════════════════════════════ # FETCH SCANS - INCREMENTAL ONLY! # ══════════════════════════════════════════════════════════════════════════════ last_fetch = load_last_fetch() fetch_time = datetime.now(timezone.utc) if last_fetch: print(f" Incremental fetch after: {last_fetch.strftime('%d-%m-%Y %I:%M %p UTC')}") query = db.collection(SCAN_COLLECTION).where( filter=firestore.FieldFilter(TIMESTAMP_FIELD, ">", last_fetch) ) else: print(" First run — full fetch...") query = db.collection(SCAN_COLLECTION) new_scans = [] for doc in query.stream(): d = doc.to_dict() if TIMESTAMP_FIELD not in d: continue ts_ist = to_ist_naive(d[TIMESTAMP_FIELD]) barcode = str(d.get("barcode","")).strip() scanned_by = str(d.get("scannedBy","")).strip() user_id = d.get("userId","") if not barcode: barcode = userid_to_badge.get(user_id,"") sewadar_info = lookup_badge(barcode) scanner_info = lookup_badge(scanned_by) new_scans.append({ "scan_id": doc.id, "userId": user_id, "badge_no": barcode, "sewadar_name": sewadar_info["name"], "sewadar_centre": sewadar_info["centre"], "satsang_point": sewadar_info["satsang_point"], "department": sewadar_info["department"], "gender": sewadar_info["gender"], "type": d.get("type",""), "scanned_at_loc": d.get("scannedAt",""), "timestamp": ts_ist.isoformat(), "date": ts_ist.strftime("%Y-%m-%d"), "time": ts_ist.strftime("%H:%M:%S"), "scanned_by_badge": scanned_by, "scanned_by_name": scanner_info["name"], "scanned_by_centre": scanner_info["centre"], }) print(f" New scans fetched: {len(new_scans)}") # ══════════════════════════════════════════════════════════════════════════════ # MERGE + CLEAN OLD SCANS # ══════════════════════════════════════════════════════════════════════════════ existing_scans = [] if os.path.exists(OUTPUT_JSON): with open(OUTPUT_JSON) as f: existing_scans = json.load(f).get("scans", []) def normalize(s): if "badge_no" not in s: s["badge_no"] = s.get("badgeNumber", s.get("barcode","")) badge = s["badge_no"] info = lookup_badge(badge) if info["name"] not in ("", "Unknown"): s["sewadar_name"] = info["name"] s["sewadar_centre"] = info["centre"] s["satsang_point"] = info["satsang_point"] s["department"] = info["department"] s["gender"] = info["gender"] scanned_by = s.get("scanned_by_badge") or s.get("scannedBy","") s["scanned_by_badge"] = scanned_by sc = lookup_badge(scanned_by) if sc["name"] not in ("", "Unknown"): s["scanned_by_name"] = sc["name"] s["scanned_by_centre"] = sc["centre"] return s # Filter to only last MAX_DAYS to keep memory small cutoff_date = (datetime.now() - timedelta(days=MAX_DAYS_KEEP)).strftime("%Y-%m-%d") existing_scans = [s for s in existing_scans if s.get("date", "") >= cutoff_date] print(f" Kept scans (last {MAX_DAYS_KEEP} days): {len(existing_scans)}") existing_ids = {s["scan_id"] for s in existing_scans} existing_scans = [normalize(s) for s in existing_scans] for s in new_scans: if s["scan_id"] not in existing_ids: existing_scans.append(s) all_scans = existing_scans print(f" Total scans in memory: {len(all_scans)}") # ══════════════════════════════════════════════════════════════════════════════ # ANALYSIS # ══════════════════════════════════════════════════════════════════════════════ sewadar_attendance = [] headcount_per_day = [] centre_summary = [] absent_per_day = [] duplicate_flags = [] missing_out = [] re_entry_flags = [] multi_scan_flags = [] all_dates = [] raw_log = [] gender_summary = [] if all_scans: df = pd.DataFrame(all_scans) df["timestamp"] = pd.to_datetime(df["timestamp"]) df["date"] = df["timestamp"].dt.strftime("%Y-%m-%d") df["time"] = df["timestamp"].dt.strftime("%H:%M:%S") df = df.sort_values("timestamp") all_dates = sorted(df["date"].unique().tolist()) # ── Per Sewadar per Date ────────────────────────────────────────────────── for badge, grp in df.groupby("badge_no"): info = lookup_badge(badge) name = info["name"] if info["name"] not in ("","Unknown") else grp["sewadar_name"].iloc[0] centre = info["centre"] if info["centre"] else grp["sewadar_centre"].iloc[0] row = { "badge_no": badge, "sewadar_name": name, "sewadar_centre": centre, "satsang_point": info["satsang_point"] or grp["satsang_point"].iloc[0], "department": info["department"] or grp["department"].iloc[0], "gender": info["gender"] or grp["gender"].iloc[0], } for date in all_dates: day = grp[grp["date"]==date] if day.empty: row[date] = {"status":"ABSENT","first_in":"-","last_out":"-"} else: ins = day[day["type"]=="IN"]["time"].tolist() outs = day[day["type"]=="OUT"]["time"].tolist() row[date] = { "status": "PRESENT", "first_in": min(ins) if ins else "-", "last_out": max(outs) if outs else "-", } sewadar_attendance.append(row) # ── Headcount per Day ───────────────────────────────────────────────────── for date in all_dates: day_df = df[df["date"]==date] headcount_per_day.append({ "date": date, "total_present": int(day_df["badge_no"].nunique()), "total_in": int(day_df[day_df["type"]=="IN"]["badge_no"].nunique()), "total_out": int(day_df[day_df["type"]=="OUT"]["badge_no"].nunique()), }) # ── Centre Summary ───────────────────────────────────────────────────────── centre_expected = {} for badge, info in badge_lookup.items(): c = info["centre"] if c: centre_expected[c] = centre_expected.get(c, 0) + 1 all_centres = sorted(df["sewadar_centre"].dropna().unique().tolist()) for centre in all_centres: if not centre: continue c_df = df[df["sewadar_centre"]==centre] c_row = { "centre": centre, "total_sewadars": int(c_df["badge_no"].nunique()), "expected": centre_expected.get(centre, 0), "dates": {} } for date in all_dates: day = c_df[c_df["date"]==date] c_row["dates"][date] = { "present": int(day["badge_no"].nunique()), "in": int(day[day["type"]=="IN"]["badge_no"].nunique()), "out": int(day[day["type"]=="OUT"]["badge_no"].nunique()), } centre_summary.append(c_row) # ── Department Expected Counts ──────────────────────────────────────────── department_expected = {} for badge, info in badge_lookup.items(): dept = info.get("department") if dept: department_expected[dept] = department_expected.get(dept, 0) + 1 # ── Absent Sewadars per Day ─────────────────────────────────────────────── for date in all_dates: scanned_badges = set(df[df["date"]==date]["badge_no"].tolist()) absent_list = [] for badge, info in badge_lookup.items(): if badge not in scanned_badges: absent_list.append({ "badge_no": badge, "name": info["name"], "centre": info["centre"], "department": info["department"], "gender": info["gender"], }) absent_per_day.append({ "date": date, "absent_count": len(absent_list), "absent": absent_list }) # ── Duplicate IN detection ──────────────────────────────────────────────── for (badge, date), grp in df.groupby(["badge_no","date"]): ins = grp[grp["type"]=="IN"] if len(ins) > 1: duplicate_flags.append({ "badge_no": badge, "sewadar_name": grp["sewadar_name"].iloc[0], "date": date, "in_count": len(ins), "times": ins["time"].tolist(), }) # ── Missing OUT ──────────────────────────────────────────────────────────── for (badge, date), grp in df.groupby(["badge_no","date"]): has_in = len(grp[grp["type"]=="IN"]) > 0 has_out = len(grp[grp["type"]=="OUT"]) > 0 if has_in and not has_out: missing_out.append({ "badge_no": badge, "sewadar_name": grp["sewadar_name"].iloc[0], "centre": grp["sewadar_centre"].iloc[0], "department": grp["department"].iloc[0], "date": date, "in_time": grp[grp["type"]=="IN"]["time"].min(), }) # ── Re-Entry Detection ──────────────────────────────────────────────────── for (badge, date), grp in df.groupby(["badge_no","date"]): day_scans = grp.sort_values("time") types = day_scans["type"].tolist() times = day_scans["time"].tolist() for i in range(len(types) - 2): if types[i] == "IN" and types[i+1] == "OUT" and types[i+2] == "IN": re_entry_flags.append({ "badge_no": badge, "sewadar_name": grp["sewadar_name"].iloc[0], "date": date, "first_in": times[i], "out_time": times[i+1], "second_in": times[i+2] }) break # ── Multi Scan Flags ────────────────────────────────────────────────────── for (badge, date), grp in df.groupby(["badge_no","date"]): ins = grp[grp["type"]=="IN"]["time"].tolist() outs = grp[grp["type"]=="OUT"]["time"].tolist() if len(ins) > 1: multi_scan_flags.append({ "badge_no": badge, "sewadar_name": grp["sewadar_name"].iloc[0], "date": date, "type": "IN", "count": len(ins), "times": sorted(ins), }) if len(outs) > 1: multi_scan_flags.append({ "badge_no": badge, "sewadar_name": grp["sewadar_name"].iloc[0], "date": date, "type": "OUT", "count": len(outs), "times": sorted(outs), }) # ── Gender Summary ──────────────────────────────────────────────────────── for date in all_dates: day_df = df[df["date"]==date] m = int(day_df[day_df["gender"]=="M"]["badge_no"].nunique()) f = int(day_df[day_df["gender"]=="F"]["badge_no"].nunique()) gender_summary.append({"date": date, "male": m, "female": f}) raw_log = df.sort_values("timestamp", ascending=False).to_dict(orient="records") # ══════════════════════════════════════════════════════════════════════════════ # EXPORT # ══════════════════════════════════════════════════════════════════════════════ def clean(obj): if isinstance(obj, float) and math.isnan(obj): return "" if isinstance(obj, dict): return {k: clean(v) for k,v in obj.items()} if isinstance(obj, list): return [clean(i) for i in obj] return obj output = clean({ "last_updated": fetch_time.strftime("%d-%m-%Y %I:%M %p UTC"), "last_scan_at": raw_log[0]["timestamp"] if raw_log else "", "total_scans": len(all_scans), "total_sewadars": len(sewadar_attendance), "total_registered": len(badge_lookup), "dates": all_dates, "scans": all_scans, "sewadar_attendance": sewadar_attendance, "headcount_per_day": headcount_per_day, "centre_summary": centre_summary, "absent_per_day": absent_per_day, "duplicate_flags": duplicate_flags, "missing_out": missing_out, "re_entry_flags": re_entry_flags, "multi_scan_flags": multi_scan_flags, "department_expected": department_expected, "gender_summary": gender_summary, "raw_log": raw_log, }) with open(OUTPUT_JSON, "w") as f: json.dump(output, f, indent=2, default=str) save_last_fetch(fetch_time) print(f" Done. {len(all_scans)} scans | {len(sewadar_attendance)} sewadars")
